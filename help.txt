using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
	public static void Main(string[] args)
	{
		int[] x= new int[]{4,5,6,2};	
		int min = 0;
		int max = 0;

		for (int i = 1; i < x.Length; i++)
		{
			if (x[min] > x[i])
				min = i;
			if (x[max] < x[i])
				max = i;
		}
		int temp = x[min];
		Console.Write(temp);
		x[min] = x[max];
		x[max] = temp;
	}
}
