            ReverseArray(arr);
        }
        static void ReverseArray<T>(T[] a)
        {
            ReverseArray(a,0,a.Length-1,a.Length/2);
        }

       static void ReverseArray<T>(T[] a,int lo,int hi,int mid)
        {
           
            if(lo > hi) return;

            // swap
            var temp = a[lo];
            a[lo] = a[hi];
            a[hi] = temp;
            Console.Write(a[hi]);
           if(mid>lo)
            Console.Write(a[lo]);
            ReverseArray(a,lo+1,hi-1,mid);
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
	public static void Main(string[] args)
	{
		int[] x= new int[]{4,5,6,2};	
		int min = 0;
		int max = 0;

		for (int i = 1; i < x.Length; i++)
		{
			if (x[min] > x[i])
				min = i;
			if (x[max] < x[i])
				max = i;
		}
		int temp = x[min];
		Console.Write(temp);
		x[min] = x[max];
		x[max] = temp;
	}
}
